    var i, tabcontent;

        tabcontent[i].style.display = "none";

function refreshData(){
   $.ajax({
       url:"/data",
       dataType:"json",
       success:function (data) {
           console.log(data);
           $("#list").html("<ul>");
           $.each(data, function (key,value) {
                $("#list").append("<li>"+value+"</li>");
           });
           $("#list").append("</ul>");
       }
   });
};
